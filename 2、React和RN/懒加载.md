import React, { useEffect, Suspense, useState } from 'react'
import { InteractionManager } from 'react-native'

export const XtaroIpollUrs = React.lazy(() => import('./XtaroIpollUrs').then((module) => ({ default: module.IUrs })))

/**
 * 懒加载组件类型
 */
type LazyComponentType<P> = React.ComponentType<P> | React.LazyExoticComponent<React.ComponentType<P>>
/**
 * 懒加载组件
 */

export function LazyComponent<P>({
    component,
    props,
    fallback = null,
}: {
    /** 组件 */
    component: LazyComponentType<P>
    /** 参数 */
    props?: P
    /** 加载前的占位组件 */
    fallback?: React.ReactNode
}): React.ReactNode {
    /** 组件加载 */
    const [Component, setComponent] = useState<LazyComponentType<P>>()
    useEffect(() => {
        const task = InteractionManager.runAfterInteractions(() => {
            setComponent(component)
        })
        return (): void => task?.cancel?.()
    }, [component])
    if (!Component) return null
    return (
        <Suspense fallback={fallback}>
            {/** @ts-expect-error Component is a valid component */}
            <Component {...props} />
        </Suspense>
    )
}
