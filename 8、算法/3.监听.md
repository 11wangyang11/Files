## 1、模拟监听
```js
// 基础事件发射器（浏览器环境）
class EventEmitter {
  constructor() {
    this.events = {}; // 事件集，使用对象方便查找
  }

  // 为指定事件event添加监听器listener，本质上就是将listener加到event的队列中
  on(event, listener) {
    if (!this.events[event]) { // 使用[]方式访问动态属性event
      this.events[event] = [];
    }
    this.events[event].push(listener);
  }

  once(event, listener) {
    // 包装新的函数，实现触发后移除监听
    const wrapper = (...args) => {
      // 先移除自身，避免递归调用导致无限循环
      this.off(event, wrapper); // 闭包，捕获到传入的event和listener
      // 执行原始监听器
      listener(...args); // 闭包
    }
    // 将新函数绑定到事件
    this.on(event, wrapper);
  }

  // 从event队列中取出所有listener执行一遍，相当于监听到了event事件触发了listener
  emit(event, ...args) {
    if (this.events[event]) {
      this.events[event].forEach(listener => listener(...args));
    }
  }

  // 移除指定事件event的特定监听器listener
  off(event, listener) {
    if (this.events[event]) {
      this.events[event] = this.events[event].filter(l => l !== listener);
    }
  }
}
```

使用示例:
```js
const emitter = new EventEmitter();

// 添加监听器
const onClick = (data) => console.log("Clicked:", data);
emitter.on("click", onClick);

// 触发事件
emitter.emit("click", "Button 1"); // 输出: Clicked: Button 1

// 移除监听器
emitter.off("click", onClick);
emitter.emit("click", "Button 2"); // 无输出
```
