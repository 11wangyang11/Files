```ts
// 代码1-
class Logger {
    private messageList: [string, number][]
    constructor() {
        this.messageList = []
    }
    shouldPrintMessage(timestamp: number, message: string): boolean {
        let rel = false
        const index = this.messageList.findIndex((msg, index) => msg[0] === message)
        if (index === -1) {
            rel = true
            this.messageList.push([message, timestamp + 10])
        } else {
            if (this.messageList[index][1] <= timestamp) {
                rel = true
                this.messageList[index][1] = timestamp + 10
            }
        }
        return rel
    }
}

// 引入Set
class Logger2 {
    private messageList: [number, string][] // 存储时间10以内的message
    private messageSet: Set<string> // 记录与messageList一致的message，用于搜索
    constructor() {
        this.messageList = []
        this.messageSet = new Set()
    }

    shouldPrintMessage(timestamp: number, message: string) {
        while(this.messageList.length > 0) {
            const [time, message] = this.messageList[0]
            if (timestamp - time >= 10) {
                this.messageList.shift()
                this.messageSet.delete(message)
            } else {
                break
            }
        }
        if (this.messageSet.has(message)) {
            return false
        }
        this.messageList.push([timestamp, message])
        this.messageSet.add(message)
        return true
    }

}
```

## 2、敲击计数器-LeetCode 362
```ts
class HitCounter {
    private hitCountList: number[]
    constructor() {
        this.hitCountList = []
    }

    hit(time: number) {
        this.hitCountList.push(time)
    }

    getHits(time: number) {
        while(this.hitCountList.length > 0) {
            const t = this.hitCountList[0]
            if (time - t >= 300) {
                this.hitCountList.shift()
            } else {
                break
            }
        }
        return this.hitCountList.length
    }
}
```

```ts
function longestSub(s: string) {
    let start = 0, end = 0
    for(let i=0;i< s.length; i++) {
        // 单
        const nums1 = getCurrentLongest(s, i, i)
        // 双
        const nums2 = getCurrentLongest(s, i, i + 1)
        const nums = nums1[1] - nums1[0] > nums2[1] - nums2[0] ? nums1 : nums2
        if (nums[1] - nums[0] > end - start) {
            end = nums[1]
            start = nums[0]
        }
        console.log(start, end)
    }
    return s.slice(start, end+1)
}

function getCurrentLongest(s: string, left: number, right: number) {
    while(left >=0 && right <= s.length-1 && s[right] === s[left]) {
        left--
        right++
    }
    return [left+1, right-1]
}
```

## 4、字母异位词分组-LeetCode 49
```ts
function groupAnagrams(strs: string[]): string[][] {
    // 判断异位词：排序后相等
    const groupMap = new Map()
    for (let i = 0; i < strs.length; i++) {
        const sortedArr = Array.from(strs[i]).sort().toString()
        if (groupMap.has(sortedArr)) {
            const list = groupMap.get(sortedArr)
            list.push(strs[i])
            groupMap.set(sortedArr, list)
        } else {
            groupMap.set(sortedArr, [strs[i]])
        }
    }
    return Array.from(groupMap.values())
};
```

```ts
function removeElement(nums: number[], val: number): number {
    let start=0, end=nums.length-1
    while(start <= end) {
        // 方法1

        if (nums[start] === val) {

            while(start <= end && nums[end] === val) {
                end--
            }
            if (start > end) break
            nums[start] = nums[end]
            end--
        }
        start++
    }
    return start
}
// 方法3:快慢指针，遍历数组，将非val值一个一个放到前面
function removeElement2(nums: number[], val: number): number {
    let slow = 0
    for(let fast = 0; fast < nums.length; fast++) {
        if(nums[fast] !== val) {
            nums[slow] = nums[fast]
            slow++ // 当前位置不是val了，就往前移动一格
        }
    }
    return slow
};
```

## 7、LRU（最近最少使用）缓存
这道题目的核心就是，Map的迭代顺序与插入顺序是一致的。Map.prototype.set 在键已存在时，只会更新值，而不会改变该键在迭代顺序中的位置。但是如果是不存在的，Map.prototype.set就会更新到队尾。所以，如果 key 存在，先取出值，然后删除该键值对，再重新插入，这样维护的map就是顺序的。通过this.cache.keys().next().value就能获取到队首的最老的元素。

当然，也可以用双向链表来维护顺序，map的value存储node便于直接访问到链表节点。双向链表则存储value还有key（因为删除的链表中老数据时也要删除map中对应的键值）。
```ts
class LRUCache {
    private cache: Map<number, number>
    private capacity: number
    constructor(capacity: number) {
        this.cache = new Map()
        this.capacity = capacity
    }

    get(key: number): number {
        if (!this.cache.has(key)) {
            return -1
        }
        // 用过就是最新的了
        const value = this.cache.get(key)
        this.cache.delete(key)
        this.cache.set(key, value)
        return value
    }

    put(key: number, value: number): void {
        // 如果 key 已存在，先删除旧记录
        if (this.cache.has(key)) {
            this.cache.delete(key)
        }
        this.cache.set(key, value)
        if(this.cache.size > this.capacity) {
            const oldKey = this.cache.keys().next().value
            this.cache.delete(oldKey)
        }
    }
}
```
